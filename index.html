<script>

const STOP_NAME = "Hamburg Behrmannplatz";


async function loadData() {

    try {

        const res = await fetch(`https://v6.db.transport.rest/stops?query=${STOP_NAME}`);

        const stops = await res.json();


        const stopId = stops[0].id;


        const depRes = await fetch(`https://v6.db.transport.rest/stops/${stopId}/departures?duration=60`);

        const data = await depRes.json();


        const table = document.getElementById("departures");


        table.innerHTML = `

            <tr>

                <th>Linie</th>

                <th>Richtung</th>

                <th class="time">Abfahrt</th>

            </tr>

        `;


        data.departures.slice(0, 8).forEach(dep => {

            const line = dep.line.name;

            const direction = dep.direction;


            const planned = new Date(dep.plannedWhen);

            const real = new Date(dep.when);


            const diffMin = Math.round((real - planned) / 60000);


            const now = new Date();

            const mins = Math.max(0, Math.round((real - now) / 60000));


            let delayHTML = "";

            let delayClass = "";


            if (diffMin > 0) {

                delayClass = diffMin > 4 ? "late" : "medium";

                delayHTML = `<span class="delay ${delayClass}">(+${diffMin})</span>`;

            }


            const timeText = mins <= 0 ? "sofort" : `${mins} min`;


            table.innerHTML += `

                <tr>

                    <td class="line">${line}</td>

                    <td>${direction}</td>

                    <td class="time">${timeText} ${delayHTML}</td>

                </tr>

            `;

        });


        document.querySelector(".footer").innerText = "Letztes Update: " + new Date().toLocaleTimeString();


    } catch (err) {

        document.querySelector(".footer").innerText = "Fehler beim Laden...";

        console.error(err);

    }

}


loadData();

setInterval(loadData, 30000);

</script>
