<script>

// Stopname

const STOP_NAME = "Behrmannplatz, Hamburg";


// CORS-freundlicher Test-Proxy

const PROXY = "https://api.allorigins.win/raw?url=";


async function load() {

    try {

        // Stop-ID holen

        const stopRes = await fetch(`${PROXY}${encodeURIComponent(`https://v6.db.transport.rest/stops?query=${STOP_NAME}`)}`);

        const stops = await stopRes.json();

        if (!stops[0]) throw new Error("Stop nicht gefunden");

        const stopId = stops[0].id;


        // Abfahrten holen

        const depRes = await fetch(`${PROXY}${encodeURIComponent(`https://v6.db.transport.rest/stops/${stopId}/departures?duration=60`)}`);

        const data = await depRes.json();


        const container = document.getElementById("list");

        container.innerHTML = "";

        const now = new Date();


        data.departures

            .filter(d => d.line.product === "bus")

            .slice(0,10)

            .forEach(dep => {

                const line = dep.line.name;

                const direction = dep.direction;


                const planned = new Date(dep.plannedWhen);

                const real = new Date(dep.when);

                const delay = Math.round((real - planned)/60000);

                const mins = Math.max(0, Math.round((real - now)/60000));


                let delayHTML = "";

                let delayClass = "";

                if(delay>0){delayClass = delay>4?"late":"medium"; delayHTML=`<span class="delay ${delayClass}">(+${delay})</span>`;}

                const timeText = mins<=0?"sofort":mins+" min";


                container.innerHTML += `

                    <div class="row">

                        <div class="line">${line}</div>

                        <div class="direction">${direction}</div>

                        <div class="time">${timeText} ${delayHTML}</div>

                    </div>

                `;

            });


        document.getElementById("status").innerText = "Letztes Update: " + new Date().toLocaleTimeString();


    } catch(e){

        document.getElementById("list").innerText = "Fehler beim Laden";

        document.getElementById("status").innerText = "";

        console.error(e);

    }

}


load();

setInterval(load,30000);

</script>
